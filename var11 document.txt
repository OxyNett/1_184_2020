#include <iostream>
#include <string>
using std::string;
using std::cout;
using std::endl;
 
class Document
{
public:
    string name;
    int size;
    int data;
    void move() {
        cout << name << " edited" << endl;
    }
};
int main()
{
    Document document;
    Document *ptr = &document;
    ptr->name = "doc1";
    ptr->size = 45;
    ptr->data = 21;
    ptr->move();
    cout << "Name: " << ptr->name << "\tSize: " << ptr->size << "\tData: " << ptr->data << endl;
    cout << "Name: " << document.name << "\tSize: " << document.size << "\tData: " << ptr->data << endl;
    cout << "Name: " << document.name << "\tSize: " << document.size << "\tData: " << ptr->data << endl;
    return 0;
}
